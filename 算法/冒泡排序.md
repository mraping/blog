/**
 * 冒泡排序
 */
public class BubbleSort {
    public static void main(String[] args) {
        int[] arr={6,3,8,2,9,1};
        for (Integer i:arr) {
            System.out.println(i);
        }

        for(int i =0;i<arr.length;i++){
            for(int j=0;j<arr.length-1-i;j++){
                if(arr[j]>arr[j+1]){
                    int tmp = arr[j+1];
                    arr[j+1] = arr[j];
                    arr[j] = tmp;
                }
            }
        }
        System.out.println("----排序后----");
        for (Integer i:arr) {
            System.out.println(i);
        }
    }
}
